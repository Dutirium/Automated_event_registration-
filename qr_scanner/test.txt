import 'dart:convert';
import 'package:flutter/material.dart';
import 'package:mobile_scanner/mobile_scanner.dart';
import 'package:http/http.dart' as http;

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return const MaterialApp(
      debugShowCheckedModeBanner: false,
      home: QRScannerPage(),
    );
  }
}

class QRScannerPage extends StatefulWidget {
  const QRScannerPage({super.key});

  @override
  State<QRScannerPage> createState() => _QRScannerPageState();
}

class _QRScannerPageState extends State<QRScannerPage> {
  final MobileScannerController controller = MobileScannerController(
    detectionSpeed: DetectionSpeed.normal,
    facing: CameraFacing.back,
  );

  bool isProcessing = false;
  String resultText = "Scan a QR";
  Color resultColor = Colors.grey;

  Future<void> verifyToken(String token) async {
    setState(() {
      resultText = "Verifying...";
      resultColor = Colors.orange;
    });

    try {
      final response = await http.post(
        Uri.parse("http://192.168.29.232/gyanJyoti/backend/verify.php"),
        headers: {"Content-Type": "application/json"},
        body: jsonEncode({"token": token}),
      );

      final data = jsonDecode(response.body);

      if (data["status"] == "valid") {
        setState(() {
          resultText = "VALID";
          resultColor = Colors.green;
        });
      } else {
        setState(() {
          resultText = "INVALID";
          resultColor = Colors.red;
        });
      }
    } catch (e) {
      setState(() {
        resultText = "NETWORK ERROR";
        resultColor = Colors.red;
      });
    }

    await Future.delayed(const Duration(seconds: 2));

    controller.start();
    isProcessing = false;
  }

  void handleDetection(BarcodeCapture capture) {
    if (isProcessing) return;

    final barcode = capture.barcodes.first;
    final raw = barcode.rawValue;

    if (raw == null) return;

    try {
      final uri = Uri.parse(raw);
      final token = uri.queryParameters['token'];

      if (token != null) {
        isProcessing = true;
        controller.stop();
        verifyToken(token);
      }
    } catch (_) {
      // If not a URL, ignore
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text("QR Entry Scanner")),
      body: Column(
        children: [
          Expanded(
            flex: 4,
            child: MobileScanner(
              controller: controller,
              onDetect: handleDetection,
            ),
          ),
          Expanded(
            flex: 1,
            child: Container(
              width: double.infinity,
              color: resultColor,
              alignment: Alignment.center,
              child: Text(
                resultText,
                style: const TextStyle(
                  fontSize: 24,
                  color: Colors.white,
                  fontWeight: FontWeight.bold,
                ),
              ),
            ),
          ),
        ],
      ),
    );
  }
}
